def *(o)
        tam = @m.length
        @result = Array.new(tam, Array.new(tam ,0))
        @fi.times do |i|
                 @co.times do |j|      
                        @fi.times do |k|
                                @result[i][j] += @m[i][k]*o[k][j]
                        end
                end
        end 
        return @result
end

def +(o)

def -(o)
        @result = @m
        @fi.times do |i|
                @co.times do |k|
                        @result[i][k] = @m[i][k] - o[i][k]
                end
        end
        return @result
end

def [](n)
        raise ArgumentError, "El parametro debe ser entero" unless n.is_a? Integer
        if (n>=0) then
                return @m[n]
        else
                return nil
        end
end

def mostrar()
        texto=""
        for i in 0...@fi do
                if (i>0) then
                        texto += "\n"
                end
                for k in 0...@co do
                        texto += "#{@m[i][k]} "
                end
        end
        texto
end
